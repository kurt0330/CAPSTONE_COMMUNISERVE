\-- \==========================================  
\-- 1\. SETUP & EXTENSIONS  
\-- \==========================================  
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

\-- \==========================================  
\-- 2\. CREATE ALL 16 TABLES   
\-- \==========================================

\-- 1\. USERS  
CREATE TABLE public.users (  
  user\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  auth\_id UUID REFERENCES auth.users(id),   
  full\_name VARCHAR(150) NOT NULL,  
  email VARCHAR(150) UNIQUE NOT NULL,  
  password\_hash VARCHAR(255) NOT NULL,   
  role TEXT CHECK (role IN ('Customer','Provider','Admin')) NOT NULL,  
  contact\_number VARCHAR(15),  
  barangay VARCHAR(100) NOT NULL,  
  municipality VARCHAR(100) NOT NULL DEFAULT 'Anini-y',  
  province VARCHAR(100) NOT NULL DEFAULT 'Antique',  
  created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,  
  updated\_at TIMESTAMP WITH TIME ZONE  
);

\-- 2\. ADMINS  
CREATE TABLE public.admins (  
  admin\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  user\_id BIGINT UNIQUE REFERENCES users(user\_id) ON DELETE CASCADE,  
  office VARCHAR(100) NOT NULL DEFAULT 'PESO',  
  position VARCHAR(100)  
);

\-- 3\. CUSTOMERS  
CREATE TABLE public.customers (  
  customer\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  user\_id BIGINT UNIQUE REFERENCES users(user\_id) ON DELETE CASCADE,  
  preferred\_barangay VARCHAR(100)  
);

\-- 4\. PROVIDERS  
CREATE TABLE public.providers (  
  provider\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  user\_id BIGINT UNIQUE REFERENCES users(user\_id) ON DELETE CASCADE,  
  trade\_category TEXT CHECK (trade\_category IN ('Carpenter','Electrician','Kasambahay','Nanny','Other')) NOT NULL,  
  admin\_status TEXT CHECK (admin\_status IN ('Pending','Approved','Rejected','Suspended')) NOT NULL DEFAULT 'Pending',  
  rejected\_at TIMESTAMP WITH TIME ZONE,  
  average\_rating DECIMAL(3,2) NOT NULL DEFAULT 0.00,  
  bio TEXT  
);

\-- 5\. ASSESSMENT TESTS  
CREATE TABLE public.assessment\_tests (  
  test\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  admin\_id BIGINT NOT NULL REFERENCES admins(admin\_id),  
  trade\_category TEXT CHECK (trade\_category IN ('Carpenter','Electrician','Kasambahay')) NOT NULL,  
  test\_title VARCHAR(150) NOT NULL,  
  passing\_score DECIMAL(5,2) NOT NULL DEFAULT 75.00,  
  is\_active BOOLEAN NOT NULL DEFAULT TRUE,  
  created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,  
  updated\_at TIMESTAMP WITH TIME ZONE,  
  UNIQUE (trade\_category, is\_active)  
);

\-- 6\. ASSESSMENT QUESTIONS  
CREATE TABLE public.assessment\_questions (  
  question\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  test\_id BIGINT NOT NULL REFERENCES assessment\_tests(test\_id) ON DELETE CASCADE,  
  question\_text TEXT NOT NULL,  
  question\_order SMALLINT NOT NULL DEFAULT 1,  
  points SMALLINT NOT NULL DEFAULT 1  
);

\-- 7\. ASSESSMENT CHOICES  
CREATE TABLE public.assessment\_choices (  
  choice\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  question\_id BIGINT NOT NULL REFERENCES assessment\_questions(question\_id) ON DELETE CASCADE,  
  choice\_text VARCHAR(300) NOT NULL,  
  is\_correct BOOLEAN NOT NULL DEFAULT FALSE,  
  choice\_order SMALLINT NOT NULL DEFAULT 1  
);

\-- 8\. ASSESSMENT ATTEMPTS  
CREATE TABLE public.assessment\_attempts (  
  attempt\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  test\_id BIGINT NOT NULL REFERENCES assessment\_tests(test\_id),  
  score\_raw DECIMAL(5,2),  
  score\_pct DECIMAL(5,2),  
  passed BOOLEAN,  
  started\_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT\_TIMESTAMP,  
  submitted\_at TIMESTAMP WITH TIME ZONE  
);

\-- 9\. ASSESSMENT ANSWERS  
CREATE TABLE public.assessment\_answers (  
  answer\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  attempt\_id BIGINT NOT NULL REFERENCES assessment\_attempts(attempt\_id) ON DELETE CASCADE,  
  question\_id BIGINT NOT NULL REFERENCES assessment\_questions(question\_id) ON DELETE CASCADE,  
  chosen\_choice\_id BIGINT REFERENCES assessment\_choices(choice\_id),  
  UNIQUE (attempt\_id, question\_id)  
);

\-- 10\. ASSESSMENTS (LGU SIDE)  
CREATE TABLE public.assessments (  
  assessment\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  admin\_id BIGINT NOT NULL REFERENCES admins(admin\_id),  
  test\_score DECIMAL(5,2),  
  lgu\_status TEXT CHECK (lgu\_status IN ('Pending','Passed','Failed')) NOT NULL DEFAULT 'Pending',  
  remarks TEXT,  
  assessed\_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT\_TIMESTAMP  
);

\-- 11\. EMPLOYMENT DETAILS  
CREATE TABLE public.employment\_details (  
  employment\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT UNIQUE NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  employment\_status TEXT CHECK (employment\_status IN ('Employed','Unemployed')) NOT NULL,  
  employment\_type TEXT CHECK (employment\_type IN ('Wage Employed','Self Employed','Not Applicable')) NOT NULL DEFAULT 'Not Applicable',  
  unemployment\_reason TEXT NOT NULL DEFAULT 'Not Applicable',  
  self\_employed\_spec VARCHAR(100),  
  highest\_education TEXT,  
  school\_last\_attended VARCHAR(150),  
  course\_completed VARCHAR(150),  
  year\_graduated VARCHAR(4),  
  employment\_history TEXT  
);

\-- 12\. JOB REQUESTS  
CREATE TABLE public.job\_requests (  
  job\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  customer\_id BIGINT NOT NULL REFERENCES customers(customer\_id) ON DELETE CASCADE,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  service\_description TEXT NOT NULL,  
  service\_street VARCHAR(150),  
  service\_barangay VARCHAR(100) NOT NULL,  
  job\_status TEXT CHECK (job\_status IN ('Pending','Accepted','Declined','Ongoing','Completed','Cancelled')) NOT NULL DEFAULT 'Pending',  
  requested\_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT\_TIMESTAMP,  
  accepted\_at TIMESTAMP WITH TIME ZONE,  
  started\_at TIMESTAMP WITH TIME ZONE,  
  completed\_at TIMESTAMP WITH TIME ZONE  
);

\-- 13\. NSRP DETAILS  
CREATE TABLE public.nsrp\_details (  
  nsrp\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT UNIQUE NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  last\_name VARCHAR(80) NOT NULL,  
  first\_name VARCHAR(80) NOT NULL,  
  middle\_name VARCHAR(80),  
  suffix VARCHAR(10),  
  date\_of\_birth DATE NOT NULL,  
  age SMALLINT NOT NULL,  
  sex TEXT CHECK (sex IN ('Male','Female')) NOT NULL,  
  civil\_status TEXT CHECK (civil\_status IN ('Single','Married','Widowed','Separated')) NOT NULL,  
  pres\_street VARCHAR(150),  
  pres\_barangay VARCHAR(100) NOT NULL,  
  pres\_city VARCHAR(100) NOT NULL,  
  pres\_province VARCHAR(100) NOT NULL,  
  perm\_street VARCHAR(150),  
  perm\_barangay VARCHAR(100),  
  perm\_city VARCHAR(100),  
  perm\_province VARCHAR(100),  
  father\_name VARCHAR(150),  
  father\_contact VARCHAR(15),  
  mother\_name VARCHAR(150),  
  mother\_contact VARCHAR(15),  
  parents\_civil\_status TEXT CHECK (parents\_civil\_status IN ('Living Together','Solo Parent','Separated')),  
  is\_4ps\_beneficiary BOOLEAN NOT NULL DEFAULT FALSE,  
  is\_indigent BOOLEAN NOT NULL DEFAULT FALSE,  
  is\_pwd BOOLEAN NOT NULL DEFAULT FALSE,  
  is\_senior\_citizen BOOLEAN NOT NULL DEFAULT FALSE,  
  is\_solo\_parent BOOLEAN NOT NULL DEFAULT FALSE  
);

\-- 14\. PROVIDER FILES  
CREATE TABLE public.provider\_files (  
  file\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  file\_type TEXT CHECK (file\_type IN ('national\_id','national\_id\_back','photo','secondary\_id','certificate')) NOT NULL,  
  file\_path VARCHAR(300) NOT NULL,  
  original\_name VARCHAR(255) NOT NULL,  
  uploaded\_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT\_TIMESTAMP  
);

\-- 15\. RATINGS  
CREATE TABLE public.ratings (  
  rating\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  job\_id BIGINT UNIQUE NOT NULL REFERENCES job\_requests(job\_id) ON DELETE CASCADE,  
  customer\_id BIGINT NOT NULL REFERENCES customers(customer\_id) ON DELETE CASCADE,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  stars SMALLINT CHECK (stars \>= 1 AND stars \<= 5\) NOT NULL,  
  review\_text TEXT,  
  rated\_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT\_TIMESTAMP  
);

\-- 16\. SKILLS  
CREATE TABLE public.skills (  
  skill\_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
  provider\_id BIGINT NOT NULL REFERENCES providers(provider\_id) ON DELETE CASCADE,  
  skill\_name VARCHAR(100) NOT NULL,  
  description VARCHAR(255),  
  years\_experience SMALLINT NOT NULL DEFAULT 0  
);

\-- \==========================================  
\-- 3\. HELPER FUNCTIONS & TRIGGERS  
\-- \==========================================

\-- Trigger Function for Ratings  
CREATE OR REPLACE FUNCTION update\_avg\_rating()  
RETURNS TRIGGER AS $$  
BEGIN  
  UPDATE providers  
  SET average\_rating \= (  
    SELECT COALESCE(ROUND(AVG(stars), 2), 0.00)  
    FROM ratings  
    WHERE provider\_id \= NEW.provider\_id  
  )  
  WHERE provider\_id \= NEW.provider\_id;  
  RETURN NEW;  
END;  
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg\_update\_avg\_rating  
AFTER INSERT OR UPDATE ON ratings  
FOR EACH ROW EXECUTE FUNCTION update\_avg\_rating();

\-- Helper function for Admin Security (MOVED HERE SO USERS TABLE EXISTS FIRST)  
CREATE OR REPLACE FUNCTION public.is\_admin()  
RETURNS BOOLEAN AS $$  
  SELECT EXISTS (  
    SELECT 1 FROM users WHERE auth\_id \= auth.uid() AND role \= 'Admin'  
  );  
$$ LANGUAGE sql SECURITY DEFINER;

\-- \==========================================  
\-- 4\. ENABLE RLS & CORE POLICIES  
\-- \==========================================

ALTER TABLE users ENABLE ROW LEVEL SECURITY;  
ALTER TABLE providers ENABLE ROW LEVEL SECURITY;  
ALTER TABLE nsrp\_details ENABLE ROW LEVEL SECURITY;

\-- User Policies  
CREATE POLICY "Users view own profile" ON users FOR SELECT USING (auth\_id \= auth.uid() OR public.is\_admin());  
CREATE POLICY "Users insert own profile" ON users FOR INSERT WITH CHECK (auth\_id \= auth.uid());  
CREATE POLICY "Users update own profile" ON users FOR UPDATE USING (auth\_id \= auth.uid());

\-- Provider Policies  
CREATE POLICY "Public sees approved providers" ON providers FOR SELECT USING (admin\_status \= 'Approved');  
CREATE POLICY "Admins view all providers" ON providers FOR SELECT USING (public.is\_admin());  
CREATE POLICY "Admins update providers" ON providers FOR UPDATE USING (public.is\_admin());

\-- Privacy Policies (NSRP)  
CREATE POLICY "Admins view NSRP" ON nsrp\_details FOR SELECT USING (public.is\_admin());
